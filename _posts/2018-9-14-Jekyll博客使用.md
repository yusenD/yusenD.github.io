### 1.语法：
---
#### js高亮
    {% highlight js %}
    // count to ten
    for (var i = 1; i <= 10; i++) {
        console.log(i);
    }
    
    // count to twenty
    var j = 0;
    while (j < 20) {
        j++;
        console.log(j);
    }
    {% endhighlight %}
